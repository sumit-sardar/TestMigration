
CREATE TABLE OAS_SCORING_EVENT_LOG
(
  INVOKE_ID             	NUMBER NOT NULL,
  ROSTER_ID             	NUMBER NOT NULL,
  STUDENT_ID            	NUMBER,
  SESSION_ID            	NUMBER,
  SCORING_EVENT         	VARCHAR2(50),
  SUBTEST_IDS           	VARCHAR2(100),
  ROSTER_VALIDATION_STATUS  VARCHAR2(2),
  INVOKE_TIMESTAMP      	DATE default SYSDATE,  
  INVOKED_BY          		VARCHAR2(50),
  MESSAGE               	VARCHAR2(500),
  CONSTRAINT INVOKE_ID_PK PRIMARY KEY (INVOKE_ID)
)   


CREATE SEQUENCE SEQ_OAS_SCORING_EVENT_LOG_ID MINVALUE 1 MAXVALUE 999999999999999999999999999 INCREMENT BY 1 START WITH 1000 CACHE 20;


CREATE INDEX OAS_SCORING_EVENT_LOG_IDX ON OAS_SCORING_EVENT_LOG(ROSTER_ID);
